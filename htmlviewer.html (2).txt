<script>
const rows = 14;   // puedes cambiar cantidad de filas
const cols = 4;

const grid = document.getElementById("grid");

function loadData(){
  const saved = localStorage.getItem("scheduleGrid");
  return saved ? JSON.parse(saved) : [];
}

function saveData(data){
  localStorage.setItem("scheduleGrid", JSON.stringify(data));
}

let data = loadData();

function createGrid(){
  for(let i=0;i<rows*cols;i++){

    if(!data[i]){
      data[i] = {
        text:"",
        color:"transparent"
      };
    }

    const cell = document.createElement("div");
    cell.classList.add("cell");
    cell.textContent = data[i].text;
    cell.style.background = data[i].color;

    cell.addEventListener("click",()=>{
      
      const newText = prompt("Editar texto:", data[i].text);
      if(newText !== null){
        data[i].text = newText;
      }

      const newColor = prompt("Color (ej: #d8c8ff o lavender):", data[i].color);
      if(newColor !== null){
        data[i].color = newColor;
      }

      cell.textContent = data[i].text;
      cell.style.background = data[i].color;

      saveData(data);
    });

    grid.appendChild(cell);
  }

  saveData(data);
}

createGrid();
</script>
